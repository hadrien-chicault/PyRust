17/// Converts a Python object to a Rust String.
18///
19/// This is a convenience wrapper around PyO3's extraction mechanism,
20/// providing a consistent interface for string conversion throughout
21/// the codebase.
22///
23/// # Arguments
24///
25/// * `obj` - The Python object to convert
26/// * `py` - Python GIL token
27///
28/// # Returns
29///
30/// The extracted string or an error if conversion fails.
31///
32/// # Errors
33///
34/// Returns a `PyErr` if the Python object cannot be converted to a string.
35///
36/// # Example
37///
38/// ```rust,ignore
39/// let py_str: PyObject = ...;
40/// let rust_str = pyobject_to_string(&py_str, py)?;
41/// ```
42///
43/// # Note
44///
45/// This function is currently unused but kept for future extensions.
46#[allow(dead_code)]
47pub fn pyobject_to_string(obj: &PyObject, py: Python) -> PyResult<String> {
48    obj.extract::<String>(py)
49}
